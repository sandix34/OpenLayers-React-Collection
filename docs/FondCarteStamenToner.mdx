OpenLayers implémente directement plusieurs fonds de carte dans son API, ainsi on dispose des sources de données tuilées suivantes : BingMaps (avec clé), Stamen, CartoDB (avec clé) et OpenStreetMap (OSM).

<FondCarteStamenToner />

```jsx
import React, { Component } from "react"

// classes nécéssaires pour afficher la carte
import Map from "ol/Map"
import View from "ol/View"
import TileLayer from "ol/layer/Tile"
import "ol/ol.css"

// Source de couche pour le serveur de tuiles Stamen.
import Stamen from "ol/source/Stamen"

class FondCarteStamenToner extends Component {
  constructor(props) {
    super(props)

    this.state = {
      center: [0, 0],
      zoom: 2,
    }

    // Stamen Toner
    this.StamenToner = new TileLayer({
      source: new Stamen({
        layer: "toner",
      }),
    })

    // Déclaration de la carte
    this.olmap = new Map({
      target: null,
      layers: [this.StamenToner],
      view: new View({
        center: this.state.center,
        zoom: this.state.zoom,
      }),
    })
  }

  componentDidMount() {
    // Défini l'id de l'élément dans lequel la carte est rendue.
    this.olmap.setTarget("map12")
  }

  render() {
    return <div id="map12" style={{ width: "100%", height: "360px" }}></div>
  }
}

export default FondCarteStamenToner
```
